# Disable the Bluetooth device in order to make the first PL011 (UART0) the
# primary UART exposed on pins 8 and 10 of the J8 header. This is the easiest
# way to get reliable serial input/output. You must indclude
# overlays/disable-bt.dtbo on the SD card.
# Ref: https://www.raspberrypi.com/documentation/computers/configuration.html#configure-uarts
dtoverlay=disable-bt

# Log start4.elf debug output to UART
uart_2ndstage=1

# Disable pull downs (required for JTAG)
gpio=22-27=np

# Enable JTAG pins GPIO22-GPIO27
enable_jtag_gpio=1
